📐 Database Normalization — 1NF → 2NF → 3NF
Unnormalized Form (UNF) — Before Normalization
-- Raw hospital data (NOT normalized)
Patient_Appt(Appt_ID, Patient_Name, Patient_Phone, Patient_City,
             Doctor_Name, Doctor_Spec, Doctor_Phone,
             Appt_Date, Services)

⚠️ Problems: Repeating groups (Services), redundant data, anomalies on insert/update/delete.

1NF — First Normal Form

Rule: Atomic values, no repeating groups, primary key defined.

-- All attributes are atomic, each row is unique
CREATE TABLE Patient_Appt_1NF (
  Appt_ID     INT PRIMARY KEY,
  Patient_ID  INT,
  Patient_Name VARCHAR(100),
  Patient_Phone VARCHAR(15),
  Doctor_Name  VARCHAR(100),
  Doctor_Spec  VARCHAR(50),
  Appt_Date    DATE,
  Service      VARCHAR(50)   -- No multi-valued, atomic
);
2NF — Second Normal Form

Rule: 1NF + No partial dependency (all non-key attributes depend on entire primary key).

-- Separate tables: no partial dependencies
CREATE TABLE Patients (
  Patient_ID   INT PRIMARY KEY,
  Patient_Name VARCHAR(100),
  Patient_Phone VARCHAR(15),
  City          VARCHAR(50)
);

CREATE TABLE Doctors (
  Doctor_ID   INT PRIMARY KEY,
  Doctor_Name VARCHAR(100),
  Specialization VARCHAR(50)
);

CREATE TABLE Appointments (
  Appt_ID    INT PRIMARY KEY,
  Patient_ID INT, -- FK → Patients
  Doctor_ID  INT, -- FK → Doctors
  Appt_Date  DATE,
  Service    VARCHAR(50)
);
3NF — Third Normal Form

Rule: 2NF + No transitive dependency (non-key attribute depends on another non-key attribute).

-- City → Pincode transitive dependency removed
CREATE TABLE Cities (
  City_ID  INT PRIMARY KEY,
  City     VARCHAR(50),
  Pincode  VARCHAR(10)
);

CREATE TABLE Patients_3NF (
  Patient_ID   INT PRIMARY KEY,
  Patient_Name VARCHAR(100),
  Phone        VARCHAR(15),
  City_ID      INT REFERENCES Cities(City_ID)
  -- Pincode now accessed via Cities, not stored here
);
Normal FormRuleAnomaly Removed
UNFRaw data, repeating groupsNone
1NFAtomic values, PK definedRepeating groups
2NFNo partial dependencyPartial dependency
3NFNo transitive dependencyTransitive dependency
⚡ Triggers & Stored Procedures
TRIGGER 1 — Auto-log on New Appointment (AFTER INSERT)
-- Automatically inserts a log entry when appointment is created
DELIMITER $$
CREATE TRIGGER after_appointment_insert
AFTER INSERT ON Appointments
FOR EACH ROW
BEGIN
  INSERT INTO Appointment_Logs
    (Appt_ID, Action, Log_Time)
  VALUES
    (NEW.Appt_ID, 'CREATED', NOW());
END$$
DELIMITER ;
TRIGGER 2 — Prevent Double Booking (BEFORE INSERT)
-- Prevents a doctor from being booked twice at same time
DELIMITER $$
CREATE TRIGGER prevent_double_booking
BEFORE INSERT ON Appointments
FOR EACH ROW
BEGIN
  DECLARE v_count INT;
  SELECT COUNT(*) INTO v_count
  FROM Appointments
  WHERE Doctor_ID = NEW.Doctor_ID
    AND Appt_Date = NEW.Appt_Date
    AND Appt_Time = NEW.Appt_Time;
  IF v_count > 0 THEN
    SIGNAL SQLSTATE '45000'
    SET MESSAGE_TEXT = 'Doctor already booked at this time!';
  END IF;
END$$
DELIMITER ;
STORED PROCEDURE — Get Patient History
-- Returns complete appointment + billing history for a patient
DELIMITER $$
CREATE PROCEDURE GetPatientHistory(IN p_id INT)
BEGIN
  SELECT
    p.Patient_Name,
    a.Appt_Date,
    d.Doctor_Name,
    d.Specialization,
    b.Amount,
    b.Payment_Status
  FROM Patients p
  JOIN Appointments a ON p.Patient_ID = a.Patient_ID
  JOIN Doctors d ON a.Doctor_ID = d.Doctor_ID
  LEFT JOIN Bills b ON p.Patient_ID = b.Patient_ID
  WHERE p.Patient_ID = p_id
  ORDER BY a.Appt_Date DESC;
END$$
DELIMITER ;

-- Call the procedure
CALL GetPatientHistory(101);
STORED PROCEDURE — Generate Monthly Bill Summary
DELIMITER $$
CREATE PROCEDURE MonthlyBillSummary(IN yr INT, IN mo INT)
BEGIN
  SELECT
    Service_Type,
    COUNT(*) AS Total_Bills,
    SUM(Amount) AS Total_Revenue,
    AVG(Amount) AS Avg_Bill
  FROM Bills
  WHERE YEAR(Bill_Date) = yr
    AND MONTH(Bill_Date) = mo
  GROUP BY Service_Type
  ORDER BY Total_Revenue DESC;
END$$
DELIMITER ;
👁️ Joins & Views
INNER JOIN — Patient + Appointment + Doctor
-- Get all appointments with patient and doctor details
SELECT
  p.Patient_ID, p.Patient_Name, p.Blood_Group,
  d.Doctor_Name, d.Specialization,
  a.Appt_Date, a.Appt_Time, a.Status
FROM Patients p
INNER JOIN Appointments a ON p.Patient_ID = a.Patient_ID
INNER JOIN Doctors d    ON a.Doctor_ID  = d.Doctor_ID
ORDER BY a.Appt_Date DESC;
LEFT JOIN — Patients with Pending Bills
-- Show all patients; mark those with unpaid bills
SELECT
  p.Patient_Name,
  COALESCE(b.Amount, 0) AS Bill_Amount,
  COALESCE(b.Payment_Status, 'No Bill') AS Pay_Status
FROM Patients p
LEFT JOIN Bills b ON p.Patient_ID = b.Patient_ID
WHERE b.Payment_Status = 'Pending'
   OR b.Patient_ID IS NULL;
VIEW — Doctor Workload Summary
-- Reusable view: number of appointments per doctor
CREATE VIEW Doctor_Workload AS
SELECT
  d.Doctor_ID,
  d.Doctor_Name,
  d.Specialization,
  COUNT(a.Appt_ID) AS Total_Appointments,
  SUM(CASE WHEN a.Status = 'Completed' THEN 1 ELSE 0 END) AS Completed
FROM Doctors d
LEFT JOIN Appointments a ON d.Doctor_ID = a.Doctor_ID
GROUP BY d.Doctor_ID, d.Doctor_Name, d.Specialization;

-- Query the view like a table
SELECT * FROM Doctor_Workload WHERE Total_Appointments > 5;
VIEW — Revenue Summary by Service
CREATE VIEW Revenue_By_Service AS
SELECT
  Service_Type,
  COUNT(*) AS Count,
  SUM(Amount)  AS Total_Revenue,
  MAX(Amount)  AS Max_Bill,
  MIN(Amount)  AS Min_Bill
FROM Bills
WHERE Payment_Status = 'Paid'
GROUP BY Service_Type;
🔄 Transactions — COMMIT & ROLLBACK
What is a Transaction?

A transaction is a sequence of SQL operations executed as a single logical unit. It follows the ACID properties:

PropertyMeaningExample
AtomicityAll or nothingBook appt + bill together or neither
ConsistencyDB stays validBalance never goes negative
IsolationTransactions don't interfereTwo doctors booked simultaneously
DurabilityCommitted data persistsSurvives power failure
TRANSACTION 1 — Book Appointment + Generate Bill (COMMIT)
START TRANSACTION;

-- Step 1: Insert appointment
INSERT INTO Appointments
  (Patient_ID, Doctor_ID, Appt_Date, Appt_Time, Status)
VALUES
  (101, 5, '2024-11-10', '10:30', 'Scheduled');

-- Step 2: Generate consultation bill
INSERT INTO Bills
  (Patient_ID, Service_Type, Amount, Payment_Status)
VALUES
  (101, 'Consultation', 500.00, 'Pending');

-- Both succeeded → commit
COMMIT;
-- ✅ Both records permanently saved
TRANSACTION 2 — Safe Payment with ROLLBACK
START TRANSACTION;

-- Step 1: Update bill to Paid
UPDATE Bills
SET Payment_Status = 'Paid', Paid_Date = NOW()
WHERE Bill_ID = 202;

-- Step 2: Insert payment record
INSERT INTO Payments(Bill_ID, Amount, Mode, Txn_Date)
VALUES(202, 1500.00, 'UPI', NOW());

-- Validate: check if amount matches
IF @@ERROR <> 0 THEN
  ROLLBACK;   -- ❌ Something failed — undo all
ELSE
  COMMIT;     -- ✅ All good — save permanently
END IF;
SAVEPOINT — Partial Rollback
START TRANSACTION;

INSERT INTO Patients ... ;           -- Step 1
SAVEPOINT after_patient;             -- 🔖 Checkpoint

INSERT INTO Appointments ... ;       -- Step 2
SAVEPOINT after_appt;

INSERT INTO Bills ... ;              -- Step 3

-- If billing fails, only undo billing
ROLLBACK TO SAVEPOINT after_appt;   -- Undo only Step 3
COMMIT;                              -- Save Steps 1 & 2
